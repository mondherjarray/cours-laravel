Exercice 1 : Création des Modèles

Créez les modèles Eloquent pour l'application TaskFlow.

1.1 Génération des modèles

Utilisez Artisan pour créer les modèles avec leurs factories et seeders :

php artisan make:model Project -fs
php artisan make:model Task -fs
php artisan make:model Category -fs
php artisan make:model Comment -fs
php artisan make:model Tag -fs
Note

L'option -fs crée le modèle avec sa Factory et son Seeder. Le modèle User existe déjà par défaut dans Laravel.

1.2 Configuration du modèle Project

Ouvrez app/Models/Project.php et configurez-le :

<?php namespace App\Models; use Illuminate\Database\Eloquent\Factories\HasFactory; use Illuminate\Database\Eloquent\Model; use Illuminate\Database\Eloquent\SoftDeletes; class Project extends Model { use HasFactory, SoftDeletes; protected $fillable = [ 'name', 'description', 'status', 'priority', 'start_date', 'due_date', 'owner_id', ]; protected $casts = [ 'start_date' => 'date', 'due_date' => 'date', 'priority' => 'integer', ]; // Les relations seront ajoutées dans l'exercice 2 }
1.3 Configuration du modèle Task

Configurez le modèle app/Models/Task.php :

<?php namespace App\Models; use Illuminate\Database\Eloquent\Factories\HasFactory; use Illuminate\Database\Eloquent\Model; use Illuminate\Database\Eloquent\SoftDeletes; class Task extends Model { use HasFactory, SoftDeletes; protected $fillable = [ 'title', 'description', 'status', 'priority', 'due_date', 'estimated_hours', 'project_id', 'assigned_to', 'category_id', ]; protected $casts = [ 'due_date' => 'date', 'priority' => 'integer', 'estimated_hours' => 'decimal:2', ]; }
  • Créer les 5 modèles avec Artisan
  • Configurer Project avec $fillable et $casts
  • Configurer Task avec $fillable et $casts
  • Configurer les modèles Category, Comment et Tag de la même manière

Exercice 2 : Définition des Relations

Définissez les relations entre les modèles selon le schéma de la base de données.

2.1 Relations du modèle User

Modifiez app/Models/User.php pour ajouter les relations :

// Dans la classe User, ajoutez : /** * Projets dont l'utilisateur est propriétaire */ public function ownedProjects() { return $this->hasMany(Project::class, 'owner_id'); } /** * Projets auxquels l'utilisateur participe */ public function projects() { return $this->belongsToMany(Project::class, 'project_user') ->withPivot('role') ->withTimestamps(); } /** * Tâches assignées à l'utilisateur */ public function assignedTasks() { return $this->hasMany(Task::class, 'assigned_to'); } /** * Commentaires de l'utilisateur */ public function comments() { return $this->hasMany(Comment::class); }
2.2 Relations du modèle Project

Ajoutez les relations dans app/Models/Project.php :

/** * Propriétaire du projet */ public function owner() { return $this->belongsTo(User::class, 'owner_id'); } /** * Membres du projet */ public function members() { return $this->belongsToMany(User::class, 'project_user') ->withPivot('role') ->withTimestamps(); } /** * Tâches du projet */ public function tasks() { return $this->hasMany(Task::class); }
2.3 Relations du modèle Task

Complétez app/Models/Task.php :

/** * Projet parent */ public function project() { return $this->belongsTo(Project::class); } /** * Utilisateur assigné */ public function assignee() { return $this->belongsTo(User::class, 'assigned_to'); } /** * Catégorie de la tâche */ public function category() { return $this->belongsTo(Category::class); } /** * Commentaires de la tâche */ public function comments() { return $this->hasMany(Comment::class); } /** * Tags de la tâche */ public function tags() { return $this->belongsToMany(Tag::class, 'task_tag'); }

  • Ajouter les relations au modèle Project
  • Ajouter les relations au modèle Task

  • Configurer Tag (belongsToMany Task)

Exercice 3 : Factories et Seeders

Créez des factories pour générer des données de test.

3.1 Factory pour Project

Modifiez database/factories/ProjectFactory.php :

<?php namespace Database\Factories; use App\Models\User; use Illuminate\Database\Eloquent\Factories\Factory; class ProjectFactory extends Factory { public function definition(): array { $startDate = fake()->dateTimeBetween('-1 month', 'now'); return [ 'name' => fake()->sentence(3), 'description' => fake()->paragraph(), 'status' => fake()->randomElement(['planning', 'active', 'on_hold', 'completed']), 'priority' => fake()->numberBetween(1, 5), 'start_date' => $startDate, 'due_date' => fake()->dateTimeBetween($startDate, '+3 months'), 'owner_id' => User::factory(), ]; } // États personnalisés public function active(): static { return $this->state(fn (array $attributes) => [ 'status' => 'active', ]); } public function completed(): static { return $this->state(fn (array $attributes) => [ 'status' => 'completed', ]); } }
3.2 Factory pour Task

Modifiez database/factories/TaskFactory.php :

<?php namespace Database\Factories; use App\Models\Project; use App\Models\User; use App\Models\Category; use Illuminate\Database\Eloquent\Factories\Factory; class TaskFactory extends Factory { public function definition(): array { return [ 'title' => fake()->sentence(4), 'description' => fake()->paragraph(), 'status' => fake()->randomElement(['todo', 'in_progress', 'review', 'done']), 'priority' => fake()->numberBetween(1, 5), 'due_date' => fake()->dateTimeBetween('now', '+2 months'), 'estimated_hours' => fake()->randomFloat(1, 1, 40), 'project_id' => Project::factory(), 'assigned_to' => User::factory(), 'category_id' => Category::factory(), ]; } }
3.3 Seeder principal

Modifiez database/seeders/DatabaseSeeder.php :

<?php namespace Database\Seeders; use App\Models\User; use App\Models\Project; use App\Models\Task; use App\Models\Category; use App\Models\Tag; use Illuminate\Database\Seeder; class DatabaseSeeder extends Seeder { public function run(): void { // Créer des catégories $categories = Category::factory()->count(5)->create(); // Créer des tags $tags = Tag::factory()->count(10)->create(); // Créer des utilisateurs $users = User::factory()->count(10)->create(); // Créer un utilisateur de test $testUser = User::factory()->create([ 'name' => 'Test User', 'email' => 'test@taskflow.com', ]); // Créer des projets avec des tâches $projects = Project::factory() ->count(5) ->create(['owner_id' => $testUser->id]); foreach ($projects as $project) { // Ajouter des membres au projet $project->members()->attach( $users->random(3)->pluck('id'), ['role' => 'developer'] ); // Créer des tâches pour le projet $tasks = Task::factory() ->count(rand(5, 10)) ->create([ 'project_id' => $project->id, 'assigned_to' => $users->random()->id, 'category_id' => $categories->random()->id, ]); // Associer des tags aux tâches foreach ($tasks as $task) { $task->tags()->attach($tags->random(rand(1, 3))); } } } }
php artisan db:seed

Exercice 5 : Scopes

Ajoutez des Query Scopes aux modèles.

5.1 Scopes pour Project

Ajoutez dans app/Models/Project.php :

use Illuminate\Database\Eloquent\Casts\Attribute; // Query Scopes public function scopeActive($query) { return $query->where('status', 'active'); } public function scopeHighPriority($query) { return $query->where('priority', '>=', 4); } public function scopeOverdue($query) { return $query->where('due_date', '<', now()) ->where('status', '!=', 'completed'); } public function scopeOwnedBy($query, $userId) { return $query->where('owner_id', $userId); }
5.2 Scopes pour Task

Ajoutez dans app/Models/Task.php :

// Scopes public function scopeTodo($query) { return $query->where('status', 'todo'); } public function scopeInProgress($query) { return $query->where('status', 'in_progress'); } public function scopeCompleted($query) { return $query->where('status', 'done'); } public function scopeAssignedTo($query, $userId) { return $query->where('assigned_to', $userId); } public function scopeDueSoon($query, $days = 7) { return $query->whereBetween('due_date', [now(), now()->addDays($days)]); }
5.3 Test dans Tinker
// Utiliser les scopes App\Models\Project::active()->get(); App\Models\Project::active()->highPriority()->get(); App\Models\Project::overdue()->count(); App\Models\Task::todo()->assignedTo(1)->get(); App\Models\Task::dueSoon(3)->get();
  • Ajouter les scopes au modèle Project
  • Ajouter les scopes au modèle Task
